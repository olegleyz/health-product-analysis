# =============================================================================
# Health Concierge — Cron Configuration
# =============================================================================
# Edit PROJECT_DIR below to match your installation path, then install with:
#   bash scripts/install_cron.sh
# =============================================================================

PROJECT_DIR=/path/to/health-concierge

# Nightly database backup — 2:00 AM
0 2 * * * cd $PROJECT_DIR && .venv/bin/python scripts/backup_db.py >> logs/backup.log 2>&1

# Data sync — every hour (with error alerting)
0 * * * * cd $PROJECT_DIR && .venv/bin/python scripts/error_alert.py .venv/bin/python scripts/sync_all.py >> logs/sync.log 2>&1

# Morning check-in — 6:00 AM (user wakes at 5:20)
0 6 * * * cd $PROJECT_DIR && .venv/bin/python scripts/error_alert.py .venv/bin/python scripts/morning_checkin.py >> logs/proactive.log 2>&1

# Evening check-in — 8:30 PM (user goes to bed 9:30-10:00)
30 20 * * * cd $PROJECT_DIR && .venv/bin/python scripts/error_alert.py .venv/bin/python scripts/evening_checkin.py >> logs/proactive.log 2>&1

# Nudge checks — every 2 hours during waking hours (6 AM - 9 PM)
0 6-21/2 * * * cd $PROJECT_DIR && .venv/bin/python scripts/error_alert.py .venv/bin/python scripts/check_nudges.py >> logs/nudges.log 2>&1

# Daily summary — 9:15 PM (before bed)
15 21 * * * cd $PROJECT_DIR && .venv/bin/python scripts/error_alert.py .venv/bin/python scripts/daily_summary.py >> logs/summary.log 2>&1

# Weekly reflection — Sunday 8:00 PM
0 20 * * 0 cd $PROJECT_DIR && .venv/bin/python scripts/error_alert.py .venv/bin/python scripts/weekly_reflection.py >> logs/proactive.log 2>&1

# Daily counter reset — midnight
0 0 * * * cd $PROJECT_DIR && .venv/bin/python scripts/error_alert.py .venv/bin/python scripts/reset_daily.py >> logs/system.log 2>&1
